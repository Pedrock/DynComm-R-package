214  bool replace(const U & oldValue, const U & newValue){
215  if(oldValue==newValue)return true;//ignore if values are the same
216  std::pair<range_iterator,range_iterator> p=r.equal_range(oldValue);//find range with old value
217  if(p.first==p.second)return false;
218  std::set<T> s;
219  for(range_iterator_const it=p.first;it!=p.second;++it){//for all keys in range
220  const std::pair<U,T> & a=*it;//get pair
221  s.insert(a.second);//get key
222  }
223  r.erase(oldValue);//erase old value reverse entry
224  for(typename std::set<T>::const_iterator it=s.cbegin();it!=s.cend();++it){//for all keys in range
225  const T & k=*it;//get key
226  m.erase(k);//erase key entry
227  m.insert(std::make_pair(k,newValue));//insert new value entry
228  r.insert(std::make_pair(newValue,k));//insert new value reverse entry
229  }
