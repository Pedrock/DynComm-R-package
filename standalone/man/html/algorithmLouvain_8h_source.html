195  void disband(const typeCommunity c1,const typeCommunity c2){
196  //remove affected communities from cg by removing all edges to their respective neighbours
197  typeLinksRangeConst nc1=cg.neighbouringCommunities(c1);
198  {
199  typeLinksIteratorConst it=nc1.first;
200  while(it!=nc1.second){
201  const typeLinksPair & p=*it;
202  if(p.first!=c1) break;
203  const HalfEdge & h=p.second;
204  ++it;
205  cg.removeEdge(c1,h.destination());
206  }
207  }
208  {
209  typeLinksRangeConst nc2=cg.neighbouringCommunities(c2);
210  typeLinksIteratorConst it=nc2.first;
211  while(it!=nc2.second){
212  const typeLinksPair & p=*it;
213  if(p.first!=c2) break;
214  const HalfEdge & h=p.second;
215  ++it;
216  cg.removeEdge(c2,h.destination());
217  }
218  }
219  //remove inner edges
220  cg.removeEdge(c1,c1);
221  cg.removeEdge(c2,c2);
222  //take nodes of affected communities from g and add them to cg disbanded by adding edges to their neighbours
223  std::set<typeVertex> ns;
224  typeCommunityListRange rc1=grph.vertices(c1);
225  for(typeCommunityListRangeIteratorConst it=rc1.first;it!=rc1.second;++it){
226  const typeCommunityListRangePair & p=*it;
227  ns.insert(p.second);
228  cg.community(p.second,p.second);//set community of vertex to vertex
229  }
230  typeCommunityListRange rc2=grph.vertices(c2);
231  for(typeCommunityListRangeIteratorConst it=rc2.first;it!=rc2.second;++it){
232  const typeCommunityListRangePair & p=*it;
233  ns.insert(p.second);
234  cg.community(p.second,p.second);//set community of node to node
235  }
236  for(std::set<typeVertex>::const_iterator it=ns.cbegin();it!=ns.cend();++it){
237  const typeVertex & n=*it;
238  typeLinksRangeConst r=grph.neighbours(n);
239  typeLinksIteratorConst itn=r.first;
240  while(itn!=r.second){
241  const typeLinksPair & p=*itn;
242  ++itn;
243  if(p.first!=n) break;
244  const HalfEdge & h=p.second;
245  const typeVertex & nei=h.destination();
246  const typeCommunity & cn=grph.community(nei);
247  if(cn!=c1 && cn!=c2){
248  cg.addEdge(n,cn,h.weight());//add edge between community of neighbour and node
249  }
250  else{
251  cg.addEdge(n,nei,h.weight());//add edge between neighbour and node
252  }
253  }
254  }
255  //disband communities on the reference graph
256  grph.disband(c1);
257  grph.disband(c2);
258  }

267  std::map<typeCommunity, typeWeight> a;

294  bool one_level(){
295  bool improvement=false ;
296  int nb_moves;
297  long double new_qual = quality();
298  long double cur_qual = new_qual;
299 
300  const typeVertexList nodes=getVertices();
301  // repeat while
302  // there is an improvement of quality
303  // or there is an improvement of quality greater than a given epsilon
304  // or a predefined number of passes have been done
305  do {
306  cur_qual = new_qual;
307  nb_moves = 0;
308 
309  // for each node: remove the node from its community and insert it in the best community
310  for (typeVertexListIteratorConst node_tmp = nodes.cbegin() ; node_tmp != nodes.cend() ; node_tmp++) {
311  const typeVertex & vertex = *node_tmp;
312  typeCommunity node_comm = community(vertex);
313 
314  // computation of all neighbour node communities of current node
315  std::map<typeCommunity, typeWeight> nc=neigh_comm(vertex);
316 
317  // compute the nearest community for node
318  // default choice for future insertion is the former community
319  typeCommunity best_comm = node_comm;
320  typeWeight best_increase = 0.0L;
321  //for all neighbours
322  for (std::map<typeCommunity, typeWeight>::iterator it=nc.begin() ; it!=nc.end() ; it++){
323  const std::pair<typeCommunity, typeWeight> & p=*it;
324  //calculate gain in quality by inserting the given node in the community of the neighbour
325  typeWeight increase=gain(vertex,p.first);
326  if (increase>best_increase) {
327  best_comm = p.first;
328  best_increase = increase;
329  }
330  }
331  // insert node in the nearest community
332  community(vertex,best_comm);
333  if (best_comm!=node_comm){
334  nb_moves++;
335  }
336  }
337 
338  new_qual = quality();
339 
340  if (nb_moves>0)
341  improvement=true;
342 
343  } while (nb_moves>0 && new_qual-cur_qual > prmtrs.precision);
344 
345  //sync changed communities back to reference graph
346  if(firstRun){
347  typeCommunities coms=grph.communities();//get all found communities
348  for(typeCommunities::const_iterator itc=coms.cbegin();itc!=coms.cend();++itc){
349  const typeCommunity & srcc=*itc;
350  //handle inner edges
351  const typeWeight & in=grph.innerEdges(srcc);
352  if(in!=0){
353  cg.addEdge(srcc,srcc,in);
354  }
355  //handle outer edges
356  typeLinksRangeConst neighbours=grph.neighbouringCommunities(srcc);
357  for(typeLinksIteratorConst itn=neighbours.first;itn!=neighbours.second;++itn){
358  const typeLinksPair & p=*itn;
359  const HalfEdge & he=p.second;
360  const typeVertex & destc=he.destination();
361  const typeWeight & weight=he.weight();
362  cg.addEdge(srcc,destc,weight);
363  }
364  }
365  firstRun=false;
366  }
367  else{//not the first run
368  typeVertexListConst coms=cg.getVertices();
369  for(typeVertexListIteratorConst itc=coms.cbegin();itc!=coms.cend();++itc){
370  const typeVertex & n=*itc;
371  const typeCommunity & c=cg.community(n);
372  if(n!=c){//community has changed
373  typeCommunityListRange r=grph.vertices(n);
374  for(typeCommunityListRangeIteratorConst itr=r.first;itr!=r.second;){
375  const typeCommunityListRangePair & p=*itr;
376  const typeVertex & nd=p.second;
377  ++itr;
378  grph.community(nd,c);
379  }
380  }
381  }
382  cg.communitiesToGraph();
383  }
384  return improvement;
