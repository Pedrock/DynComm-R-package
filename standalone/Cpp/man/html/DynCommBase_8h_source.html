263  bool communityMapping(WriterInterface * writer,bool communityFirst=true,bool differential=true) const{
264  if(differential){
265  if(communityFirst){
266  const typeCommunities & gc=grph.communities();
267  for(typeCommunities::const_iterator it=gc.cbegin();it!=gc.cend();++it){
268  const typeCommunity &c=*it;
269  typeVertexList n=vertices(c);
270  typeVertexList changed=typeVertexList();//empty set of vertices that changed community
271  //check vertices that existed or are new
272  for(typeVertexListIteratorConst itn=n.cbegin();itn!=n.cend();++itn){
273  const typeVertex & v=*itn;
274  const typeCommunityListIteratorConst & itc=oldCommunities.value(v);
275  if(itc==oldCommunities.cend()){//vertex did not exist so consider it changed community
276  changed.insert(v);
277  }
278  else{//vertex existed
279  const typeCommunity & oc=(*itc).second;
280  if(c!=oc){//changed community
281  changed.insert(v);
282  }
283  }
284  }
285  if(changed.size()>0){
286  unsigned int i=1;
287  writer->write(std::to_string(c),WriterInterface::WRITETYPE::VALUE);
288  for(typeVertexListIteratorConst itn=changed.cbegin();itn!=changed.cend();++itn){
289  const typeVertex & v=*itn;
290  if(i==changed.size()){//last value
291  writer->write(std::to_string(v),WriterInterface::WRITETYPE::LINE);
292  }
293  else{
294  writer->write(std::to_string(v),WriterInterface::WRITETYPE::VALUE);
295  }
296  ++i;
297  }
298  }
299  //check vertices that no longer exist
300  changed.clear();
301  if(grph.vertexCount()!=oldCommunities.size()){//some vertices no longer exist
302  const typeVertexList & vs=grph.getVertices();
303  for(typeCommunityListIteratorConst itn=oldCommunities.cbegin();itn!=oldCommunities.cend();++itn){
304  const typeVertex & v=(*itn).first;
305  if(vs.find(v)==vs.end()){//vertex no longer exists so consider it changed community
306  changed.insert(v);
307  }
308  }
309  }
310  if(changed.size()>0){
311  unsigned int i=1;
312  writer->write("NA",WriterInterface::WRITETYPE::VALUE);
313  for(typeVertexListIteratorConst itn=changed.cbegin();itn!=changed.cend();++itn){
314  const typeVertex & v=*itn;
315  if(i==changed.size()){//last value
316  writer->write(std::to_string(v),WriterInterface::WRITETYPE::LINE);
317  }
318  else{
319  writer->write(std::to_string(v),WriterInterface::WRITETYPE::VALUE);
320  }
321  ++i;
322  }
323  }
324  }
325  }
326  else{//vertex first
327  //check vertices that existed or are new
328  const typeVertexList & vs=grph.getVertices();
329  for(typeVertexListIteratorConst it=vs.cbegin();it!=vs.cend();++it){
330  const typeVertex & v=*it;
331  const typeCommunity & c=grph.community(v);
332  const typeCommunityListIteratorConst itc=oldCommunities.value(v);
333  if(itc==oldCommunities.cend()){//vertex did not exist so consider it changed community
334  writer->write(std::to_string(v),WriterInterface::WRITETYPE::VALUE);
335  writer->write(std::to_string(c),WriterInterface::WRITETYPE::LINE);
336  }

338  const typeCommunity & oc=(*itc).second;
339  if(c!=oc){//changed community
340  writer->write(std::to_string(v),WriterInterface::WRITETYPE::VALUE);
341  writer->write(std::to_string(c),WriterInterface::WRITETYPE::LINE);
342  }
343  }
344  }
345  //check vertices that no longer exist

347  for(typeCommunityListIteratorConst itn=oldCommunities.cbegin();itn!=oldCommunities.cend();++itn){
348  const typeVertex & v=(*itn).first;

360  for(typeCommunities::const_iterator it=gc.cbegin();it!=gc.cend();++it){
361  const typeCommunity &c=*it;

363  typeVertexList n=vertices(c);

367  writer->write(std::to_string(*itn),WriterInterface::WRITETYPE::LINE);
